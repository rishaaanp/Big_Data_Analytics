5. Order data (DESC) - 0: jdbc:hive2://localhost:10000/default> SELECT * FROM sales ORDER BY category ASC, price DESC;

9. Combined sum - 0: jdbc:hive2://localhost:10000/default> SELECT category, SUM(price * quantity) AS total_revenue FROM sales GROUP BY category;
10. Min Max - 0: jdbc:hive2://localhost:10000/default> SELECT category, MAX(price) AS max_price, MIN(price) AS min_price FROM sales GROUP BY category;
11. Combined - 0: jdbc:hive2://localhost:10000/default> SELECT category, SUM(price * quantity) AS total_revenue FROM sales GROUP BY category, ORDER BY total_revenue DESC;
